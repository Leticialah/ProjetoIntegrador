cmake_minimum_required(VERSION 3.19)
project(AutoAtendimento LANGUAGES CXX)

include(GNUInstallDirs)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core Widgets SerialPort LinguistTools)

qt_standard_project_setup()

qt_add_executable(AutoAtendimento
    WIN32 MACOSX_BUNDLE
    main.cpp
    mainwindow.cpp
    mainwindow.h
    mainwindow.ui
    ESP32Manager.cpp
    ESP32Manager.h
)

qt_add_translations(
    TARGETS AutoAtendimento
    TS_FILES AutoAtendimento_pt_BR.ts
)

target_link_libraries(AutoAtendimento
    PRIVATE
        Qt::Core
        Qt::Widgets
        Qt::SerialPort
)

install(TARGETS AutoAtendimento
    BUNDLE  DESTINATION .
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)

qt_generate_deploy_app_script(
    TARGET AutoAtendimento
    OUTPUT_SCRIPT deploy_script
    NO_UNSUPPORTED_PLATFORM_ERROR
)
install(SCRIPT ${deploy_script})
